const STELLAR_NETWORK = process.env.STELLAR_NETWORK || "testnet"; // "testnet" | "public"
const HORIZON_URL =

/* ===========================
   SIMPLE IN-MEMORY LEDGER
   (mint/stake/burn for your API)
   =========================== */

let ledger = {
  totalSupply: 0,
  users: {}, // { userId: { balance: number, staked: number } }
};

function ensureUser(userId) {
  if (!ledger.users[userId]) ledger.users[userId] = { balance: 0, staked: 0 };
  return ledger.users[userId];
}

export function getTotalSupply() {
  return ledger.totalSupply;
}

export function getUserInfo(userId) {
  return ensureUser(userId);
}

export function mint(userId, amount) {
  const n = Number(amount);
  if (!Number.isFinite(n) || n <= 0) throw new Error("Invalid amount");

  const user = ensureUser(userId);
  user.balance += n;
  ledger.totalSupply += n;

  return { status: "Minted", user: userId, newBalance: user.balance };
}

export function stake(userId, amount) {
  const n = Number(amount);
  if (!Number.isFinite(n) || n <= 0) throw new Error("Invalid amount");

  const user = ensureUser(userId);
  if (user.balance < n) throw new Error("Insufficient balance");

  user.balance -= n;
  user.staked += n;

  return { status: "Staked", user: userId, balance: user.balance, staked: user.staked };
}

export function burn(userId, amount) {
  const n = Number(amount);
  if (!Number.isFinite(n) || n <= 0) throw new Error("Invalid amount");

  const user = ensureUser(userId);
  if (user.balance < n) throw new Error("Insufficient balance to burn");

  user.balance -= n;
  ledger.totalSupply -= n;

  return { status: "Burned", user: userId, remainingBalance: user.balance };
}

export function createWallet() {
  const kp = Keypair.random();
  return { publicKey: kp.publicKey(), secret: kp.secret() };
}

export async function fundTestnetAccount(publicKey) {
  if (STELLAR_NETWORK !== "testnet") {
    throw new Error("fundTestnetAccount only works on testnet (Friendbot).");
  }
  const url = `https://friendbot.stellar.org?addr=${encodeURIComponent(publicKey)}`;
  const r = await fetch(url);
  const body = await r.text();
  if (!r.ok) throw new Error(`Friendbot failed (${r.status}): ${body}`);
  return JSON.parse(body);
}

export async function getWalletBalances(publicKey) {
  const acct = await server.loadAccount(publicKey);
  return acct.balances.map((b) => ({
    asset_type: b.asset_type,
    asset_code: b.asset_code || "XLM",
    asset_issuer: b.asset_issuer || null,
    balance: b.balance,
  }));
}

function requireValidIssuer(issuerPublicKey) {
  if (
    !issuerPublicKey ||
    typeof issuerPublicKey !== "string" ||
    issuerPublicKey.length !== 56 ||
    !issuerPublicKey.startsWith("G")
  ) {
    throw new Error(
      `Invalid issuer public key: must be a 56-char Stellar G... address`
    );
  }
}

export async function trustAsset(walletSecret, assetCode, issuerPublicKey, limit = undefined) {
  if (!assetCode || typeof assetCode !== "string") throw new Error("Invalid assetCode");
  requireValidIssuer(issuerPublicKey);

  const kp = Keypair.fromSecret(walletSecret);
  const publicKey = kp.publicKey();

  // This will 404 if the account isn't funded yet:
  const account = await server.loadAccount(publicKey);

  const asset = new Asset(assetCode, issuerPublicKey);

  const op = Operation.changeTrust({
    asset,
    ...(limit ? { limit: String(limit) } : {}),
  });

  const tx = new TransactionBuilder(account, {
    fee: String(BASE_FEE),
    networkPassphrase: NETWORK_PASSPHRASE,
  })
    .addOperation(op)
    .setTimeout(180)
    .build();

  tx.sign(kp);
  return await server.submitTransaction(tx);
}

/**
 * Convenience trust for CVT using env vars:
 * CVT_CODE (default "CVT")
 * CVT_ISSUER_PUBLIC (required)
 */
export async function trustCVT(walletSecret) {
  const code = process.env.CVT_CODE || "CVT";
  const issuer = process.env.CVT_ISSUER_PUBLIC;
  if (!issuer) throw new Error("Missing env CVT_ISSUER_PUBLIC (issuer G... public key)");
  return trustAsset(walletSecret, code, issuer);
}

export async function issueAsset(issuerSecret, assetCode, destinationPublicKey, amount) {
  if (!amount) throw new Error("Amount required");
  if (!destinationPublicKey || destinationPublicKey.length !== 56) {
    throw new Error("Invalid destination public key");
  }

  const issuerKp = Keypair.fromSecret(issuerSecret);
  const issuerPub = issuerKp.publicKey();

  const issuerAccount = await server.loadAccount(issuerPub);

  const asset = new Asset(assetCode, issuerPub);

  const tx = new TransactionBuilder(issuerAccount, {
    fee: String(BASE_FEE),
    networkPassphrase: NETWORK_PASSPHRASE,
  })
    .addOperation(
      Operation.payment({
        destination: destinationPublicKey,
        asset,
        amount: String(amount),
      })
    )
    .setTimeout(180)
    .build();

  tx.sign(issuerKp);
  return await server.submitTransaction(tx);
}

import fs from "fs";
import path from "path";
import { fileURLToPath } from "url";

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);

const assetsDir = path.join(__dirname, "assets");

// ------------------------------------------------------------
// Load all asset files from /config/assets/
// ------------------------------------------------------------
export function loadAssets() {
  const registry = [];

  try {
    const files = fs.readdirSync(assetsDir);

    for (const file of files) {
      if (!file.endsWith(".json")) continue;

      const fullPath = path.join(assetsDir, file);
      const raw = fs.readFileSync(fullPath, "utf8");

      try {
        const asset = JSON.parse(raw);

        // Basic validation
        if (!asset.symbol || !asset.name) {
          console.warn(`⚠️ Invalid asset file skipped: ${file}`);
          continue;
        }

        // Enforce CVT monetary law
        if (asset.symbol === "CVT") {
          if (!asset.internal_floor || asset.internal_floor !== 1.0) {
            console.warn("⚠️ CVT internal floor corrected to 1.0 USD");
            asset.internal_floor = 1.0;
          }
          if (!asset.fortress_rules) {
            asset.fortress_rules = {};
          }
          asset.fortress_rules.never_below_floor = true;
        }

        registry.push(asset);
      } catch (err) {
        console.error(`Failed to parse asset file ${file}:`, err);
      }
    }

    return registry;
  } catch (err) {
    console.error("Failed to load asset directory:", err);
    return [];
  }
}

import express from "express";
import { alfredChat } from "./engine.js";

const router = express.Router();

const ALFRED_API_KEY = process.env.ALFRED_API_KEY || "";
const ALFRED_FOUNDER_KEY = process.env.ALFRED_FOUNDER_KEY || "";
const ALLOW_NON_LOCAL_ALFRED = (process.env.ALLOW_NON_LOCAL_ALFRED || "").toLowerCase() === "true";

function isLocalRequest(req) {
  const ip = req.ip || "";
  return ip === "127.0.0.1" || ip === "::1" || ip === "::ffff:127.0.0.1";
}

function requireAccess(req, res) {
  // If an API key is set, require it (local or not).
  if (ALFRED_API_KEY) {
    const provided = req.header("X-ALFRED-KEY") || "";
    if (!provided || provided !== ALFRED_API_KEY) {
      res.status(401).json({ error: "unauthorized", message: "Missing/invalid X-ALFRED-KEY" });
      return false;
    }
    return true;
  }

  // Otherwise: local-only unless explicitly allowed.
  if (!ALLOW_NON_LOCAL_ALFRED && !isLocalRequest(req)) {
    res.status(403).json({
      error: "forbidden",
      message: "Alfred is localhost-only. Set ALFRED_API_KEY or ALLOW_NON_LOCAL_ALFRED=true if you intend to expose it."
    });
    return false;
  }

  return true;
}

function requireFounder(req, res) {
  const provided = req.header("X-FOUNDER-KEY") || "";
  if (!ALFRED_FOUNDER_KEY || provided !== ALFRED_FOUNDER_KEY) {
    res.status(401).json({ error: "unauthorized", message: "Missing/invalid X-FOUNDER-KEY" });
    return false;
  }
  return true;
}

    return res.json({ ok: true, requestId: out.requestId, response: out.text });
  } catch (e) {
    return res.status(500).json({ ok: false, error: "internal_error", message: e.message });

    const context = req.body?.context || {};

// © 2026 The Fortress Technologies. All rights reserved.

import express from "express";

const router = express.Router();

// Security posture
const UFOS_API_KEY = process.env.UFOS_API_KEY || "";
const ALLOW_NON_LOCAL_UFOS = (process.env.ALLOW_NON_LOCAL_UFOS || "").toLowerCase() === "true";

function isLocalRequest(req) {
  const ip = (req.ip || "").trim();
  return ip === "127.0.0.1" || ip === "::1" || ip === "::ffff:127.0.0.1";
}

/**
 * Private access gate:
 * - If UFOS_API_KEY is set, require X-UFOS-KEY for private endpoints (local or not).
 * - Otherwise: local-only unless ALLOW_NON_LOCAL_UFOS=true.
 */
function requirePrivateAccess(req, res) {
  if (UFOS_API_KEY) {
    const provided = req.header("X-UFOS-KEY") || "";
    if (!provided || provided !== UFOS_API_KEY) {
      res.status(401).json({ error: "unauthorized", message: "Missing/invalid X-UFOS-KEY" });
      return false;
    }
    return true;
  }

  if (!ALLOW_NON_LOCAL_UFOS && !isLocalRequest(req)) {

/**
 * Each CMD must default-export an object:
 * {
 *   id: "stellar",
 *   supports: ["send","swap"...],
 *   execute(utr): Promise<receipt>
 * }
 */
export async function loadCMD(chain) {
  const safe = String(chain).toLowerCase().replace(/[^a-z0-9_-]/g, "");
  const modPath = path.join(CMD_DIR, `${safe}.cmd.js`);

  try {
    const mod = await import(pathToFileURL(modPath).href);
    if (!mod?.default?.execute) throw new Error("CMD missing default export with execute()");
    return mod.default;
  } catch (err) {
    throw new Error(`UFOS: Unable to load CMD for chain="${chain}". (${err.message})`);
  }
}
